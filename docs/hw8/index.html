Code 
library(vegawidget)
library(readr)
library(dplyr)
library(tidyr)

gas <- read_csv("https://calvin-data304.netlify.app/data/pump_price_for_gasoline_us_per_liter.csv")
codes <- read_csv("https://raw.githubusercontent.com/lukes/ISO-3166-Countries-with-Regional-Codes/master/all/all.csv")

gas_recent <- gas |>
  pivot_longer(-country, names_to = "year", values_to = "price_per_liter",
               values_transform = list(price_per_liter = as.numeric)) |>
  filter(!is.na(price_per_liter)) |>
  group_by(country) |>
  filter(year == max(year)) |>
  ungroup() |>
  mutate(price_per_gallon = round(price_per_liter * 3.785, 2))

plot_data <- gas_recent |>
  left_join(codes, by = c("country" = "name")) |>
  filter(!is.na(`alpha-3`), !is.na(region)) |>
  select(country, code = `alpha-3`, region, price_per_gallon)

as_vegaspec(list(
  `$schema` = "https://vega.github.io/schema/vega-lite/v5.json",
  title = list(
    text = "Gas Prices Around the World (USD per gallon)",
    subtitle = "Most recent year available. Each dot is one country.",
    anchor = "start"
  ),
  data = list(values = plot_data),
  width = 500,
  height = 300,
  mark = list(type = "point", filled = TRUE, opacity = 0.7),
  encoding = list(
    x = list(
      field = "price_per_gallon",
      type = "quantitative",
      title = "Price (USD per gallon)",
      axis = list(format = "$,.2f")
    ),
    y = list(
      field = "region",
      type = "nominal",
      title = NULL,
      sort = list(field = "price_per_gallon", op = "median", order = "descending")
    ),
    color = list(
      field = "region",
      type = "nominal",
      legend = NULL
    ),
    tooltip = list(
      list(field = "country", type = "nominal", title = "Country"),
      list(field = "code", type = "nominal", title = "Code"),
      list(field = "price_per_gallon", type = "quantitative", title = "USD/gallon", format = "$.2f")
    )
  )
))

This strip plot shows that European and Oceanian countries tend to pay more for gas while Asian and African countries show much more variation, with some paying very little, likely due to government subsidies. The Americas cluster in the middle range.
